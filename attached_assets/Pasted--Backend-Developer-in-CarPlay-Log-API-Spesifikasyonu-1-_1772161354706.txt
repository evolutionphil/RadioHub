{
  _id: ObjectId,

  logs: [

5. Örnek Python/FastAPI Implementasyonu
from fastapi import APIRouter, Query
from datetime import datetime, timedelta
from typing import Optional

router = APIRouter(prefix="/api/logs")

@router.get("/remote")
async def get_remote_logs(
    limit: int = Query(50, le=500),
    platform: Optional[str] = None,
    level: Optional[str] = None,
    search: Optional[str] = None,
    deviceId: Optional[str] = None,
    from_date: Optional[datetime] = None,
    to_date: Optional[datetime] = None,
):
    query = {}
    
    if platform:
        query["platform"] = platform
    if deviceId:
        query["deviceId"] = deviceId
    if level:
        query["logs.level"] = level
    if search:
        query["logs.message"] = {"$regex": search, "$options": "i"}
    if from_date or to_date:
        query["createdAt"] = {}
        if from_date:
            query["createdAt"]["$gte"] = from_date
        if to_date:
            query["createdAt"]["$lte"] = to_date
    
    logs = await db.remote_logs.find(
        query,
        {"_id": 0}
    ).sort("createdAt", -1).limit(limit).to_list(limit)
    
    return {
        "success": True,
        "count": len(logs),
        "logs": logs
    }


@router.get("/remote/stats")
async def get_log_stats():
    today = datetime.utcnow().replace(hour=0, minute=0, second=0)
    
    pipeline = [
        {
            "$facet": {
                "total": [{"$count": "count"}],
                "today": [
                    {"$match": {"createdAt": {"$gte": today}}},
                    {"$count": "count"}
                ],
                "byPlatform": [
                    {"$group": {"_id": "$platform", "count": {"$sum": 1}}}
                ],
                "byLevel": [
                    {"$unwind": "$logs"},
                    {"$group": {"_id": "$logs.level", "count": {"$sum": 1}}}
                ],
                "carplayConnected": [
                    {"$unwind": "$logs"},
                    {"$match": {"logs.message": {"$regex": "CarPlay CONNECTED"}}},
                    {"$count": "count"}
                ],
                "carplayErrors": [
                    {"$unwind": "$logs"},
                    {"$match": {
                        "logs.message": {"$regex": "CarPlay|Template"},
                        "logs.level": "error"
                    }},
                    {"$count": "count"}
                ]
            }
        }
    ]
    
    result = await db.remote_logs.aggregate(pipeline).to_list(1)
    # ... format and return stats


@router.delete("/remote")
async def clear_old_logs(older_than_days: int = Query(30, ge=1)):
    cutoff = datetime.utcnow() - timedelta(days=older_than_days)
    result = await db.remote_logs.delete_many({"createdAt": {"$lt": cutoff}})
    
    return {
        "success": True,
        "deletedCount": result.deleted_count,
        "message": f"Logs older than {older_than_days} days deleted"
    }
